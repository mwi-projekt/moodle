-- Course-Module-ID und Activity-ID ermitteln:
SELECT
    cm.id AS coursemoduleid,
    d.id AS dhbwioid,
    d.name
FROM mdl_course_modules cm
JOIN mdl_modules m ON m.id = cm.module
JOIN mdl_dhbwio d ON d.id = cm.instance
WHERE m.name = 'dhbwio';

-- Dataform-ID ermitteln:
SELECT
    df.id AS dataid,
    df.course
FROM mdl_dhbwio_dataform df;
